<script>
    let currentMode = ''; // 'hutang' atau 'piutang'

    function showSection(id, title) {
        document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));
        
        if(id === 'form-hutang' || id === 'form-piutang') {
            currentMode = id.includes('hutang') ? 'hutang' : 'piutang';
            document.getElementById('form-section').classList.add('active');
            document.getElementById('form-title').innerText = title;
            document.getElementById('label-kas').innerText = currentMode === 'hutang' ? 'Simpan ke' : 'Diambil dari';
        } else if(id === 'data-hutang' || id === 'data-piutang') {
            currentMode = id.includes('hutang') ? 'hutang' : 'piutang';
            document.getElementById('data-section').classList.add('active');
            document.getElementById('data-title').innerText = title;
            renderTable();
        } else {
            document.getElementById('menu-utama').classList.add('active');
        }
    }

    function simpanData() {
        const data = {
            tgl: document.getElementById('tgl').value,
            ket: document.getElementById('ket').value,
            nilai: document.getElementById('jumlah').value,
            bayar: 0
        };

        let storageKey = currentMode === 'hutang' ? 'db_hutang' : 'db_piutang';
        let list = JSON.parse(localStorage.getItem(storageKey)) || [];
        list.push(data);
        localStorage.setItem(storageKey, JSON.stringify(list));
        
        alert('Data berhasil disimpan!');
        showSection('menu-utama');
    }

    function renderTable() {
        let storageKey = currentMode === 'hutang' ? 'db_hutang' : 'db_piutang';
        let list = JSON.parse(localStorage.getItem(storageKey)) || [];
        let html = '';
        list.forEach(item => {
            html += `<tr>
                <td>${item.tgl}</td>
                <td style="text-align:left; color:blue;">${item.ket}</td>
                <td>${Number(item.nilai).toLocaleString()}</td>
                <td>${Number(item.bayar).toLocaleString()}</td>
            </tr>`;
        });
        document.getElementById('table-body').innerHTML = html;
    }
</script>
